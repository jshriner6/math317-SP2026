  \title{Math 317: Chapter 3 Portfolio}
  \author{Your name here}
  \maketitle
  \begin{enumerate}
      \item[Problem 3.2] Prove that there exists a unique additive identity of $\mathbb{R}$.
      \item[Solution 3.2]

      \item[Problem 3.3] Prove that there exists a unique multiplicative identity of $\mathbb{R}$.
      \item[Solution 3.3]

      \item[Problem 3.4] Prove that every real number has a unique additive inverse.
      \item[Solution 3.4]

      \item[Problem 3.5] Prove that every nonzero real number has a unique multiplicative inverse.
      \item[Solution 3.5]

      \item[Problem 3.8] Prove that for all $a,b,c \in \mathbb{R}$, we have the following:
          \begin{enumerate}
              \item $a=b$ if and only if $a+c = b+c$;
              \item $0a=0$;
              \item $-a = (-1)a$;
              \item $(-1)^2 = 1$;
              \item $-(-a) = a$;
              \item If $a \neq 0$, then $(a^{-1})^{-1}=a$;
              \item If $a \neq 0$ and $ab=ac$, then $b=c$.
              \item If $ab=0$, then either $a=0$ or $b=0$.
          \end{enumerate}
      \item[Solution 3.8]

      \item[Problem 3.9] Carefully prove that for all $a,b \in \mathbb{R}$, we have $(a+b)(a-b) = a^2 - b^2$. Explicitly cite where you are utilizing the Field Axioms and Problem 3.8.
      \item[Solution 3.9]

      \item[Problem 3.13] Prove that for all $a,b \in \mathbb{R}$, if $a,b \gt 0$, then $a+b \gt 0$; and if $a,b \lt 0$, then $a+b \lt 0$.
      \item[Solution 3.13]

      \item[Problem 3.14] Prove that for all $a,b,c,d \in \mathbb{R}$, if $a \lt b$ and $c \lt d$, then $a+c \lt b + d$.
      \item[Solution 3.14]

      \item[Problem 3.15] Prove that for all $a \in \mathbb{R}$, $a \gt 0 $ if and only if $-a \lt 0$.
      \item[Solution 3.15]

      \item[Problem 3.16] Prove that if $a,b,c$, and $d$ are all positive real numbers such that $a \lt b$ and $c \lt d$, then $ac \lt bd$.
      \item[Solution 3.16]

      \item[Problem 3.17] Prove that for all $a,b \in \mathbb{R}$, we have the following:
          \begin{enumerate}
              \item $ab \gt 0$ if and only if either $a,b \gt 0$ or $a,b \lt 0$;
              \item $ab \lt 0$ if and only if $a\lt 0 \lt b$ or $b \lt 0 \lt a$.
          \end{enumerate}
      \item[Solution 3.17]

      \item[Problem 3.18] Prove that for all positive real numbers $a$ and $b$, $a \lt b$ if and only if $a^2 \lt b^2$.
      \item[Solution 3.18]

      \item[Problem 3.19] Prove that for all $a \in \mathbb{R}$, we have $a^2 \geq 0$.
      \item[Solution 3.19]

      \item[Problem 3.20] Prove that $0 \lt 1$.
      \item[Solution 3.20]

      \item[Problem 3.21] Prove that for all $a \in \mathbb{R}$, if $a \gt 0$, then $a^{-1} \gt 0$, and if $a \lt 0$, then $a^{-1} \lt 0$.
      \item[Solution 3.21]

      \item[Problem 3.22] Prove that for all $a,b \in \mathbb{R}$, if $a \lt b$, then $-b \lt -a$.
      \item[Solution 3.22]

      \item[Problem] Prove that if $a$ and $b$ are both positive or both negative with $a \lt b$, then $b^{-1} \lt a^{-1}$.
      \item[Solution]

      \item[Problem 3.23] Prove that for all $a,b,c \in \mathbb{R}$, if $a \lt b$ and $c \lt 0$, then $bc \lt ac$.
      \item[Solution 3.23]

      \item[Problem 3.25] Prove that for all $a \in \mathbb{R}$, $|a| \geq 0$ with equality only if $a=0$.
      \item[Solution 3.25]

      \item[Problem 3.26] Prove that for all $a,b \in \mathbb{R}$, we have $|a-b| = |b-a|$.
      \item[Solution 3.25]

      \item[Problem 3.27] Prove that for all $a,b \in \mathbb{R}$, $|ab| = |a||b|$.
      \item[Solution 3.27]

      \item[Problem 3.28] Prove that for all $a,b \in \mathbb{R}$, if $\pm ab \leq b$, then $|a| \leq b$.
      \item[Solution 3.28]

      \item[Problem 3.29] Prove that for all $a \in \mathbb{R}$, $|a|^2 = a^2$.
      \item[Solution 3.29]

      \item[Problem 3.30] Prove that for all $a \in \mathbb{R}$, $\pm a \leq |a|$.
      \item[Solution 3.30]

      \item[Problem 3.31] Prove that for all $a,r \in \mathbb{R}$ with $r$ nonnegative, $|a|\leq r$ if and only if $-r \leq a \leq r$.
      \item[Solution 3.31]

      \item[Problem 3.32] Prove that for all $a,b,r \in \mathbb{R}$ with $r$ nonnegative, $|a-b| \leq r$ if and only if $b-r \leq a \leq b+r$.
      \item[Solution 3.32]

      \item[Problem 3.33] Prove that for all $a,b \in \mathbb{R}$, $|a+b| \leq |a| +|b|$. 
      \item[Solution 3.33]

      \item[Problem 3.34] Under what conditions do we have equality for the Triangle Inequality?
      \item[Solution 3.34]

      \item[Problem 3.35] Prove that for all $a,b \in \mathbb{R}$, $|a-b| \geq ||a| -|b||$. 
      \item[Solution 3.35]
  \end{enumerate}
