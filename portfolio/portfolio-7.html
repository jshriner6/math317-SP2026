       \item[Problem 6.3] Prove each of the following real functions is continuous using Definition 6.2.
       \begin{enumerate}
           \item $f:\mathbb{R} \to \mathbb{R}$ defined via $f(x)=x$.
           \item $g: \mathbb{R} \to \mathbb{R}$ defined via $g(x)=2x$.
           \item $h: \mathbb{R} \to \mathbb{R}$ defined via $h(x)=x+3$.
       \end{enumerate}
       \item[Solution 6.3]

       \item[Problem 6.4] Prove that every linear function $f:\mathbb{R} \to \mathbb{R}$ given by $f(x) = mx + b$ is continuous.

       \item[Solution 6.5]

       \item[Problem 6.6] Find an example of a real function $f$ that satisfies each of the following:
           \begin{enumerate}
               \item[(i)] $1 \in \text{Dom}(f)$;
               \item[(ii)] For any open interval $J$ containing $1$, there is an open interval $I$ containing $f(1)$ such that if $x \in \text{Dom}(f) \cap J$, then $f(x) \in I$;
               \item[(iii)] $f$ is not continuous at $1$.
           \end{enumerate}
       \item[Solution 6.6]

       \item[Problem 6.7] Suppose $f$ is a real function such that $a \in \text{Dom}(f)$. Prove that $f$ is continuous at $a$ if and only if for every $\varepsilon \gt 0$, there exists $\delta \gt 0$ such that if $x \in \text{Dom}(f)$ and $|x-a| \lt \delta$, the $|f(x)-f(a)| \lt \varepsilon$.
       \item[Solution 6.7]

       \item[Problem 6.8] Draw a figure in the spirit of Figure 6.1 that captures the essence of Problem 6.7.

                       \end{cases}

Prove that $\displaystyle \lim_{x\to a}f(x)$ exists if and only if for every sequence $(p_n)$ in $\text{Dom}(f) \setminus\{a\}$ converging to $a$, the sequence $(f(p_n))$ converges, in which case, $\displaystyle \lim_{x\to a}f(x)$ equals the limit of the sequence $(f(p_n))$.
       \item[Solution 7.12]

       \item[Problem 7.13] Let $f,g:A \to \mathbb{R}$ be real functions. Prove each of the following using Definition 7.1 or Problem 7.12.
           \begin{enumerate}
               \item If $c \in \mathbb{R}$, then $\displaystyle \lim_{x\to a}c = c$.
               \item If $\displaystyle \lim_{x\to a}f(x)$ and $\displaystyle \lim_{x\to a}g(x)$ both exist, then
               \[
               \lim_{x\to a}(f(x) \pm g(x)) = \lim_{x\to a}f(x) \pm \lim_{x\to a}g(x).
               \]
               \item If $\displaystyle \lim_{x\to a}f(x)$ and $\displaystyle \lim_{x\to a}g(x)$ both exist, then
               \[
               \lim_{x\to a}(f(x) \cdot g(x)) = \lim_{x\to a}f(x) \cdot \lim_{x\to a}g(x).
               \]
               \item If $c \in \mathbb{R}$ and $\displaystyle \lim_{x\to a}f(x)$ exists, then
               \[
               \lim_{x\to a}(c \cdot f(x)) = c \cdot\lim_{x\to a}f(x).
               \]
               \item If $\displaystyle \lim_{x\to a}f(x)$ and $\displaystyle \lim_{x\to a}g(x)$ both exist and $\displaystyle \lim_{x\to a}g(x) \neq 0$, then
               \[
               \lim_{x\to a}\frac{f(x)}{ g(x)} = \frac{\lim_{x\to a}f(x) }{ \lim_{x\to a}g(x)}.
               \]
               \item If $f$ is continuous at $b$ and $\displaystyle \lim_{x\to a}g(x) =b$, then
               \[
               \lim_{x\to a}f(g(x)) = f(\lim_{x\to a}g(x)) = f(b).
               \]
           \end{enumerate}
       \item[Solution 7.13]
